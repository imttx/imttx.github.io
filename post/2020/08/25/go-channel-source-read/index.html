 26type hchan struct {
 27	qcount   uint           // chan 里元素数量
 28	dataqsiz uint           // chan 底层环形队列的大小

 46// sudog 表示了一个等待队列中的 g，例如在一个 channel 中进行发送和接受
 47// sudog 是必要的，因为 g <-> 同步对象之间的关系是多对多。一个 g 可以在多个等待列表上，
 48// 因此可以有很多的 sudog 为一个 g 服务；并且很多 g 可能在等待同一个同步对象，
 49// 因此也会有很多 sudog 为一个同步对象服务。
 50// 所有的 sudog 分配在一个特殊的池中。使用 acquireSudog 和 releaseSudog 来分配并释放它们。
 51type sudog struct {
 52	// 下面的字段由这个 sudog 阻塞的通道的 hchan.lock 进行保护。
 53	// shrinkstack 依赖于它服务于 sudog 相关的 channel 操作。
 54
 55	g           *g             // 当前 goroutine
 56	next        *sudog         // 后一个 sudog
 57	prev        *sudog         // 前一个 sudog
 58	elem        unsafe.Pointer // 数据元素
 59	acquiretime int64          // 获取时间
 60	releasetime int64          // 释放时间
 61	ticket      uint32         // 信号量中使用
 62	isSelect    bool           // isSelect 表示 g 正在参与一个 select，因此 g.selectDone 必须以 CAS 的方式来避免唤醒时候的 data race。
 63	parent      *sudog         // semaRoot 二叉树
 64	waitlink    *sudog         // g.waiting 列表或 semaRoot
 65	waittail    *sudog         // semaRoot
 66	c           *hchan         // 当前 channel
 67}

 84// 通用初始化 channel 方法
 85func makechan(t *chantype, size int) *hchan {
 86	elem := t.elem
 87
 88	// 编译器检查
 89	if elem.size >= 1<<16 {
 90		throw("makechan: invalid channel element type")
 91	}
 92	if hchanSize%maxAlign != 0 || elem.align > maxAlign {
 93		throw("makechan: bad alignment")
 94	}
 95
 96	// 计算存放数据元素的内存大小以及是否溢出
 97	mem, overflow := math.MulUintptr(elem.size, uintptr(size))
 98	if overflow || mem > maxAlloc-hchanSize || size < 0 {
 99		panic(plainError("makechan: size out of range"))
100	}
101
102	var c *hchan
103	switch {
104	case mem == 0: // 不带缓存，只需要申请 hchanSize 大小的内存即可
105		c = (*hchan)(mallocgc(hchanSize, nil, true)) // 申请内存
106		c.buf = c.raceaddr()                         // race 检查使用此位置进行同步
107	case elem.ptrdata == 0: // 存放的元素不包含指针，只需要申请 hchanSize+mem 大小的内存即可
108		c = (*hchan)(mallocgc(hchanSize+mem, nil, true)) // 申请内存
109		c.buf = add(unsafe.Pointer(c), hchanSize)        // 计算 buf 指向的位置
110	default: // 存放的元素包含指针，这里需要两次申请内存
111		c = new(hchan)
112		c.buf = mallocgc(mem, elem, true)
113	}
114
115	c.elemsize = uint16(elem.size)   // 元素大小赋值
116	c.elemtype = elem                // 元素类型赋值
117	c.dataqsiz = uint(size)          // 缓存大小赋值
118	lockInit(&c.lock, lockRankHchan) // 初始化锁
119
120	return c
121}

123// chanbuf 获取 i 索引位置对应的缓存数据，类似 c 里面的指针操作
124func chanbuf(c *hchan, i uint) unsafe.Pointer {
125	return add(c.buf, uintptr(i)*uintptr(c.elemsize))

128// full reports whether a send on c would block (that is, the channel is full).
129// It uses a single word-sized read of mutable state, so although
130// the answer is instantaneously true, the correct answer may have changed
131// by the time the calling function receives the return value.
132func full(c *hchan) bool {
133	// c.dataqsiz is immutable (never written after the channel is created)
134	// so it is safe to read at any time during channel operation.
135	if c.dataqsiz == 0 {
136		// Assumes that a pointer read is relaxed-atomic.
137		return c.recvq.first == nil
138	}
139	// Assumes that a uint read is relaxed-atomic.
140	return c.qcount == c.dataqsiz
141}

149// channel 发送数据
150func chansend(c *hchan, ep unsafe.Pointer, block bool, callerpc uintptr) bool {
151	if c == nil { // 判断 channel 是否为 nil
152		if !block { // 如果非阻塞，直接返回 false
153			return false
154		}
155
156		// 当向 nil channel 发送数据时，会调用 gopark
157		// 而 gopark 会将当前的 goroutine 休眠，并用过第一个参数的 unlockf 来回调唤醒
158		// 但此处传递的参数为 nil，因此向 channel 发送数据的 goroutine 和接收数据的 goroutine 都会阻塞，进而死锁
159		gopark(nil, nil, waitReasonChanSendNilChan, traceEvGoStop, 2)
160		throw("unreachable")
161	}
162
163	// 忽略 race 检查...
164
165	// 对于不阻塞的 send，快速检测失败场景
166	// 如果 channel 未关闭且 channel 没有多余的缓冲空间。这可能是：
167	// 1. channel 是非缓冲型的，且等待接收队列里没有 goroutine
168	// 2. channel 是缓冲型的，但循环数组已经装满了元素
169	// 主要用于 select 语句中，涉及到指令重排队+可观测性，不展开讲...
170	if !block && c.closed == 0 && full(c) {
171		return false
172	}
173
174	// 加锁
175	lock(&c.lock)
176
177	// 再次检查 channel 是否已关闭，不允许向关闭的 channel 发送数据
178	if c.closed != 0 {
179		unlock(&c.lock)
180		panic(plainError("send on closed channel"))
181	}
182
183	// 从 recvq 队首取出一个接受者，如果接受者不是 nil，就绕过环形队列直接把 ep 拷贝给 sg，并释放锁
184	if sg := c.recvq.dequeue(); sg != nil {
185		send(c, sg, ep, func() { unlock(&c.lock) }, 3)
186		return true
187	}
188
189	// 如果环形队列还未满
190	if c.qcount < c.dataqsiz {
191		// 拿到 sendx 索引的位置
192		qp := chanbuf(c, c.sendx)
193
194		// 直接把数据从 qp 拷贝到 qp
195		typedmemmove(c.elemtype, qp, ep)
196		c.sendx++
197		if c.sendx == c.dataqsiz {
198			c.sendx = 0
199		}
200		c.qcount++
201		unlock(&c.lock)
202		return true
203	}
204
205	// 非阻塞，直接返回错误
206	if !block {
207		unlock(&c.lock)
208		return false
209	}
210
211	gp := getg()           // 获取当前 goroutine
212	mysg := acquireSudog() // 从对象池获取 sudog
213	mysg.elem = ep         // 保存 send 数据
214	mysg.g = gp            // 当前协程
215	mysg.isSelect = false  // 是否在 select 中
216	mysg.c = c             // 保存当前 channel
217	gp.waiting = mysg      // 保存当前 sudog，下面要用到做校验
218	gp.param = nil         // 这里为 nil，唤醒时保存传递的参数
219	c.sendq.enqueue(mysg)  // 存入队列
220
221	// 挂起当前的 g，将当前的 g 移出调度器的队列
222	gopark(chanparkcommit, unsafe.Pointer(&c.lock), waitReasonChanSend, traceEvGoBlockSend, 2)
223
224	// 当 g 被重新调度时，从这里继续往下执行
225	KeepAlive(ep)
226
227	// 检验是否为当前的 sudog
228	if mysg != gp.waiting {
229		throw("G waiting list is corrupted")
230	}
231	gp.waiting = nil
232
233	// 正常唤醒时 param 保存的是传递的参数，为 nil 时则直接 panic
234	if gp.param == nil {
235		// 如果 channel 已被关闭，则为虚假唤醒
236		if c.closed == 0 {
237			throw("chansend: spurious wakeup")
238		}
239		panic(plainError("send on closed channel"))
240	}
241	// 参数置空
242	gp.param = nil
243
244	// 取消与之前阻塞的 channel 的关联
245	mysg.c = nil
246
247	// 将 sudog 放回对象池
248	releaseSudog(mysg)
249	return true
250}

252// send 函数处理向一个空的 channel 发送操作
253// ep 指向被发送的元素，会被直接拷贝到接收的 goroutine
254// 之后，接收的 goroutine 会被唤醒
255// c 必须是空的（因为等待队列里有 goroutine，肯定是空的）
256// c 必须被上锁，发送操作执行完后，会使用 unlockf 函数解锁
257// sg 必须已经从等待队列里取出来了
258// ep 必须是非空，并且它指向堆或调用者的栈
259func send(c *hchan, sg *sudog, ep unsafe.Pointer, unlockf func(), skip int) {
260	// 忽略 race 检查
261	if sg.elem != nil {
262		sendDirect(c.elemtype, sg, ep) // 直接拷贝到接受者内存，使用写屏障
263		sg.elem = nil
264	}
265	gp := sg.g
266	unlockf()
267	gp.param = unsafe.Pointer(sg)
268	// 唤醒接收者的 goroutine. skip 和打印栈相关，先忽略
269	goready(gp, skip+1)
270}
271
272// 向一个非缓冲型的 channel 发送数据、从一个无元素的（非缓冲型或缓冲型但空）的 channel
273// 接收数据，都会导致一个 goroutine 直接操作另一个 goroutine 的栈
274// 由于 GC 假设对栈的写操作只能发生在 goroutine 正在运行中并且由当前 goroutine 来写
275// 所以这里实际上违反了这个假设。可能会造成一些问题，所以需要用到写屏障来规避
276func sendDirect(t *_type, sg *sudog, src unsafe.Pointer) {
277	// src 在当前 goroutine 的栈上，dst 是另一个 goroutine 的栈
278	// 直接进行内存"搬迁"
279	// 如果目标地址的栈发生了栈收缩，当我们读出了 sg.elem 后
280	// 就不能修改真正的 dst 位置的值了
281	// 因此需要在读和写之前加上一个屏障
282	dst := sg.elem
283	typeBitsBulkBarrier(t, uintptr(dst), uintptr(src), t.size)
284	// 拷贝内存
285	memmove(dst, src, t.size)
286}

288func recvDirect(t *_type, sg *sudog, dst unsafe.Pointer) {
289	// dst is on our stack or the heap, src is on another stack.
290	// The channel is locked, so src will not move during this
291	// operation.
292	src := sg.elem
293	typeBitsBulkBarrier(t, uintptr(dst), uintptr(src), t.size)
294	memmove(dst, src, t.size)
295}

297// 关闭 channel
298func closechan(c *hchan) {
299	// 判断 channel 是否为 nil
300	if c == nil {
301		panic(plainError("close of nil channel"))
302	}
303
304	// 加锁保护
305	lock(&c.lock)
306
307	// 判断 channel 是否已关闭，重复关闭报 panic
308	if c.closed != 0 {
309		unlock(&c.lock)
310		panic(plainError("close of closed channel"))
311	}
312
313	// 关闭 channel
314	c.closed = 1
315
316	// gList 是通过 g.schedlink 链接 G 的列表，一个 G 只能是一次在一个 gQueue 或 gList 上
317	// gList 模拟的是栈操作（FILO）
318	// gQueue 模拟的是队列操作（FIFO）
319	var glist gList
320
321	// 释放所有接收者
322	for {
323		sg := c.recvq.dequeue()
324		// sg == nil，表示接收队列已为空，跳出循环
325		if sg == nil {
326			break
327		}
328
329		// 如果 elem 不为空说明未忽略接收值，赋值为该类型的零值
330		if sg.elem != nil {
331			typedmemclr(c.elemtype, sg.elem)
332			sg.elem = nil
333		}
334		gp := sg.g
335		gp.param = nil
336		glist.push(gp) // 放入 glist 队列
337	}
338
339	// 释放所有的发送者（它们将会 panic）
340	for {
341		sg := c.sendq.dequeue()
342		if sg == nil {
343			break
344		}
345		sg.elem = nil
346		gp := sg.g
347		gp.param = nil
348		glist.push(gp)
349	}
350
351	// 释放锁
352	unlock(&c.lock)
353
354	// 循环读取 glist 里面的数据
355	for !glist.empty() {
356		gp := glist.pop()
357		gp.schedlink = 0
358		// 唤醒 goroutine 等待调度
359		goready(gp, 3)
360	}
361}

363// empty reports whether a read from c would block (that is, the channel is
364// empty).  It uses a single atomic read of mutable state.
365func empty(c *hchan) bool {
366	// c.dataqsiz is immutable.
367	if c.dataqsiz == 0 {
368		return atomic.Loadp(unsafe.Pointer(&c.sendq.first)) == nil
369	}
370	return atomic.Loaduint(&c.qcount) == 0
371}

386// chanrecv 函数接收 channel c 的元素并将其写入 ep 所指向的内存地址。
387// 如果 ep 是 nil，说明忽略了接收值。
388// 如果 block == false，即非阻塞型接收，在没有数据可接收的情况下，返回 (false, false)
389// 否则，如果 c 处于关闭状态，将 ep 指向的地址清零，返回 (true, false)
390// 否则，用返回值填充 ep 指向的内存地址。返回 (true, true)
391// 如果 ep 非空，则应该指向堆或者函数调用者的栈
392func chanrecv(c *hchan, ep unsafe.Pointer, block bool) (selected, received bool) {
393	// 忽略 debug ...
394
395	if c == nil {
396		if !block {
397			return
398		}
399		gopark(nil, nil, waitReasonChanReceiveNilChan, traceEvGoStop, 2)
400		throw("unreachable")
401	}
402
403	// 这块主要用在 select 语句中，先大概了解下，比较难懂。。。
404	// 快速路径: 在不需要锁的情况下检查失败的非阻塞操作
405	// 注意到 channel 不能由已关闭转换为未关闭，则失败的条件是：
406	// 1. channel 是非缓冲型的，recvq 队列为空
407	// 2. channel 是缓冲型的，buf 为空
408	if !block && empty(c) {
409		// 此处的 c.closed 必须在条件判断之后进行验证，
410		if atomic.Load(&c.closed) == 0 {
411			// 因为指令重排后，如果先判断 c.closed，得出 channel 未关闭，无法判断失败条件中
412			// channel 是已关闭还是未关闭（从而需要 atomic 操作）
413			return
414		}
415
416		// 再次检查 channel 是否为空
417		if empty(c) {
418			// 接受者不为 nil 时返回该类型的零值
419			if ep != nil {
420				typedmemclr(c.elemtype, ep)
421			}
422			// （true,fasle）里面的 true：表示被 select case 选中，fasle 表示是否正常收到数据
423			return true, false
424		}
425	}
426
427	// 加锁
428	lock(&c.lock)
429
430	// 处理 channel 关闭 && 无缓存元素的情况
431	if c.closed != 0 && c.qcount == 0 {
432		// race 检查...
433		unlock(&c.lock)
434		if ep != nil { // 从已关闭的 channel 继续接收值
435			typedmemclr(c.elemtype, ep) // 清理该值的内存数据，返回该类型的零值
436		}
437		return true, false
438	}
439
440	// 发送队列不为空
441	if sg := c.sendq.dequeue(); sg != nil {
442		recv(c, sg, ep, func() { unlock(&c.lock) }, 3)
443		return true, true
444	}
445
446	// 如果环形队列不为空，直接从队列接收数据
447	if c.qcount > 0 {
448		qp := chanbuf(c, c.recvx) // 获取索引位置
449		if ep != nil {
450			typedmemmove(c.elemtype, ep, qp) //内存拷贝
451		}
452		typedmemclr(c.elemtype, qp) // 内存数据清理
453		c.recvx++
454		if c.recvx == c.dataqsiz {
455			c.recvx = 0
456		}
457		c.qcount--
458		unlock(&c.lock)
459		return true, true
460	}
461
462	// 如果是非阻塞，直接返回 fasle, false
463	if !block {
464		unlock(&c.lock)
465		return false, false
466	}
467
468	// 读取不到数据，就阻塞当前 goroutine
469	gp := getg()
470	mysg := acquireSudog()
471	mysg.elem = ep
472	mysg.g = gp
473	gp.waiting = mysg
474	mysg.isSelect = false
475	mysg.c = c
476	gp.param = nil
477	c.recvq.enqueue(mysg) // 写入接收队列
478	// 挂起当前 goroutine，使其不再被调度
479	gopark(chanparkcommit, unsafe.Pointer(&c.lock), waitReasonChanReceive, traceEvGoBlockRecv, 2)
480
481	// 唤醒后，继续往下执行
482
483	// 检查是否是当前 fgoroutine
484	if mysg != gp.waiting {
485		throw("G waiting list is corrupted")
486	}
487	// 清理数据
488	gp.waiting = nil
489	closed := gp.param == nil
490	gp.param = nil
491	mysg.c = nil
492
493	// 将 sudog 放回对象池
494	releaseSudog(mysg)
495	return true, !closed
496}

498// 等待发送队列里有 goroutine 存在，说明 buf 是满的
499// 这有可能是：
500// 1. 非缓冲型的 channel
501// 2. 缓冲型的 channel，但 buf 满了
502// 针对 1，直接进行内存拷贝
503// 针对 2，接收到循环数组头部的元素，并将发送者的元素放到循环数组尾部
504func recv(c *hchan, sg *sudog, ep unsafe.Pointer, unlockf func(), skip int) {
505	// 非缓冲型的 channel
506	if c.dataqsiz == 0 {
507		if ep != nil {
508			// 直接进行内存拷贝
509			recvDirect(c.elemtype, sg, ep)
510		}
511	} else {
512		// 缓冲型的 channel
513		qp := chanbuf(c, c.recvx)
514
515		// 从队列 recvx 位置拷贝数据到接收者
516		if ep != nil {
517			typedmemmove(c.elemtype, ep, qp)
518		}
519		// 从发送者把数据写入 recvx，然后修改 recvx 位置
520		typedmemmove(c.elemtype, qp, sg.elem)
521		c.recvx++
522		if c.recvx == c.dataqsiz {
523			c.recvx = 0
524		}
525		c.sendx = c.recvx // c.sendx = (c.sendx+1) % c.dataqsiz
526	}
527	sg.elem = nil
528	gp := sg.g
529	unlockf()
530	gp.param = unsafe.Pointer(sg)
531	// 唤醒发送者 goroutine
532	goready(gp, skip+1)
533}

535func chanparkcommit(gp *g, chanLock unsafe.Pointer) bool {
536	// There are unlocked sudogs that point into gp's stack. Stack
537	// copying must lock the channels of those sudogs.
538	gp.activeStackChans = true
539	unlock((*mutex)(chanLock))
540	return true
541}

649// 放回对象池
650func (q *waitq) enqueue(sgp *sudog) {
651	sgp.next = nil
652	x := q.last
653	if x == nil {
654		sgp.prev = nil
655		q.first = sgp
656		q.last = sgp
657		return
658	}
659	sgp.prev = x
660	x.next = sgp
661	q.last = sgp
662}
663
664// 从对象池取出 sudog
665func (q *waitq) dequeue() *sudog {
666	for {
667		sgp := q.first
668		if sgp == nil {
669			return nil
670		}
671		y := sgp.next
672		if y == nil {
673			q.first = nil
674			q.last = nil
675		} else {
676			y.prev = nil
677			q.first = y
678			sgp.next = nil // mark as removed (see dequeueSudog)
679		}
680
681		// 这里涉及到 select，可先忽略这块。。。
682		//
683		// 如果一个 goroutine 由于一个 select 被放到此队列，则在 goroutine 唤醒前
684		// 如果由于 select 而将 goroutine 放在此队列中，则在由不同的情况唤醒的 goroutine 之间
685		// 存在一个小窗口并且它获取 channel lock。 一旦锁定，它会将自己从队列中删除，
686		// 因此我们不会在那之后看到它。
687		// 我们在 G 结构中使用一个标志来告诉我们其他人何时获得了这个竞争条件以告知这个 goroutine，
688		// 但是 goroutine 还没有将自己从队列中移除。
689		// 正在参与一个 select，如果将 g 标记为已完成失败，则说明此 sgp 已经不在队列中，继续循环并重新出队
690		if sgp.isSelect && !atomic.Cas(&sgp.g.selectDone, 0, 1) {
691			continue
692		}
693
694		return sgp
695	}
696}
